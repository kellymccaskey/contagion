\title{Economic Ties that Bind}
\subtitle{Economic Integration and its Influence on Civil War Contagion}
\author{Kelly McCaskey} 
\institute{University at Buffalo, SUNY}
\date{April 29th, 2015} 

\frame{\titlepage} 

\frame{\frametitle{The question}
\begin{itemize}
\item If economic integration can bring interstate peace, can it also bring intrastate peace? \pause 
\item Not so much.
\end{itemize} 
}

\frame{\frametitle{Geographic proximity}
\begin{itemize}
\item Negative consequences are rarely confined to the state experiencing war\pause
\item Transnational ties \pause
\item Economic effects 
\end{itemize} 
}

\frame{\frametitle{Interstate peace through interdependence}
\begin{itemize}
\item Economic ties reflect the economic cost of war \pause
\item Measure of the value of maintaining the status quo \pause
\item Indicator of compatibility \pause
\item Cheap talk
\end{itemize} 
}

\frame{\frametitle{Intrastate peace through interdependence?}
\begin{itemize}
\item Inhibited economic growth \pause
\item Indicator of compatibility\pause
\item The cost of rebellion 
\end{itemize} 
}

\frame{\frametitle{Hypothesis}
\begin{itemize}
\item States who trade heavily with their civil war-experiencing neighbor are more likely to also experience a civil war than those that do not. 
\end{itemize} 
}

\frame{\frametitle{Data + model}
\begin{itemize}
\item 2603 civil war-neighbor state-years from 1949 to 2006 \pause
\item DV:
\begin{itemize}
\item Civil war onset
\begin{itemize}
\item Uppsala/PRIO Conflict Data Set \pause
\end{itemize}
\end{itemize}
\item IV:
\begin{itemize}
\item Trade flows
\begin{itemize}
\item Imports from neighbor as a percentage of total imports
\item Exports to neighbor as a percentage of total exports
\item COW Trade Data \pause
\end{itemize}
\item Neighbor experiencing civil war
\begin{itemize}
\item Uppsala/PRIO Conflict Data Set 
\item COW Contiguity Data\pause
\end{itemize}
\end{itemize}
\item MLE w/ 2 logit models
\end{itemize}
}
